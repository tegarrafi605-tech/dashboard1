async function fetchData() { // Karena UI/Path sudah Anda set terpisah, kita langsung tarik fileName dari UI const fileName = document.getElementById('sumberData').value; document.getElementById('loadingIndicator').classList.remove('hidden'); document.getElementById('loadingIndicator').classList.add('flex'); document.getElementById('dashboardContent').classList.add('hidden'); try { // Fetch file selalu dieksekusi agar menjamin data ter-refresh sempurna // (Menghapus cache checking agar "nyangkut" di data lama tidak terjadi) const response = await fetch(fileName); if (!response.ok) throw new Error("File tidak ditemukan"); const arrayBuffer = await response.arrayBuffer(); workbookCache = XLSX.read(arrayBuffer, { type: 'array' }); // Langsung proses data processWorkbook(); } catch(e) { console.error(e); alert("Gagal memuat " + fileName + ". Pastikan path file benar dan tersedia."); document.getElementById('loadingIndicator').classList.remove('flex'); document.getElementById('loadingIndicator').classList.add('hidden'); } } function processWorkbook() { if(!workbookCache) return; const categoryName = document.getElementById('sumberData').options[document.getElementById('sumberData').selectedIndex].text; // DIRECT DATA ACCESS: Secara paksa & otomatis ambil Sheet Index ke-0 (Sheet Pertama) // Mengabaikan nama sheet sama sekali agar file Sains/Literasi bisa langsung terbuka. const selectedSheet = workbookCache.SheetNames[0]; if(!selectedSheet) { alert(`Tidak ada sheet yang valid di dalam file ${categoryName}.`); document.getElementById('loadingIndicator').classList.add('hidden'); document.getElementById('dashboardContent').classList.add('hidden'); document.getElementById('subtitleText').innerHTML = `Data tidak tersedia.`; return; } document.getElementById('subtitleText').innerHTML = `Menampilkan data ${categoryName} dari File: ${categoryName}`; const worksheet = workbookCache.Sheets[selectedSheet]; let rawData = XLSX.utils.sheet_to_json(worksheet, { defval: "" }); // DETEKSI KOLOM SKOR OTOMATIS targetMetricCol = null; if(rawData.length > 0) { const keys = Object.keys(rawData[0]); let scoresByCol = {}; keys.forEach(k => scoresByCol[k] = 0); rawData.forEach(row => { keys.forEach(k => { if (!isNaN(parseFloat(row[k])) && String(row[k]).trim() !== "") scoresByCol[k]++; }); }); let maxN = 0; keys.forEach(k => { if (scoresByCol[k] > maxN) { maxN = scoresByCol[k]; targetMetricCol = k; } }); } // CLEAN MAPPING & STRICT GARBAGE FILTERING let cleanData = []; if(rawData.length > 0 && targetMetricCol) { const countryKey = Object.keys(rawData[0])[0]; cleanData = rawData.filter(row => { const countryVal = String(row[countryKey] || '').trim(); const valL = countryVal.toLowerCase(); if(!countryVal || valL === 'undefined' || valL === 'null') return false; // Buang baris yang bukan negara (Teks judul, TOC, dll) const garbages = ['table', 'toc', 'note', 'figure', 'empty', 'unknown', 'oecd average', 'oecd', 'partners', 'mean']; if(garbages.some(g => valL.includes(g))) return false; if(countryVal.length > 40) return false; // Buang baris jika kolom skornya tidak berisi angka const metricVal = row[targetMetricCol]; if(metricVal === "" || isNaN(parseFloat(metricVal))) return false; return true; // Hanya loloskan Negara dan Angka Riil }); } // Urutkan data dari terbesar ke terkecil cleanData.sort((a, b) => parseFloat(b[targetMetricCol]) - parseFloat(a[targetMetricCol])); maxValueGlobal = 0; if(cleanData.length > 0) { maxValueGlobal = parseFloat(cleanData[0][targetMetricCol]); } // REFRESH TOTAL: Gambar ulang grafik dan tabel secara real-time renderCharts(cleanData); renderTable(cleanData); document.getElementById('loadingIndicator').classList.remove('flex'); document.getElementById('loadingIndicator').classList.add('hidden'); document.getElementById('dashboardContent').classList.remove('hidden'); document.getElementById('dashboardContent').classList.add('flex'); }
